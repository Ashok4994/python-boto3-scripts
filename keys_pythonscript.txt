 filters = [{
            'Name': 'instance-state-name', 
            'Values': ['running']
        },
          {'Name': 'tag:Role', 
            'Values': ['MyRoleTagValue']}
        
    ]

#Making sure that there are actually instances to shut down.

	shuttingDown = ec2.instances.filter(InstanceIds=runningInstances).stop()
	print shuttingDown
